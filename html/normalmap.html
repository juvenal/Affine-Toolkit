surface
tebNormalMap()
{
   point  Nf;  /* face forward normal */

   Nf = point faceforward( normalize(N), I );

   Ci = color( (xcomp(Nf)+1)/2, (ycomp(Nf)+1)/2, -zcomp(Nf) );
}
